  3 /**
  4  * @class Ext.ux.layout.RowFitLayout
  5  * @extends Ext.layout.ContainerLayout
  6  * <p>Layout that distributes heights of elements so they take 100% of the
  7  * container height.</p>
  8  * <p>Height of the child element can be given in pixels (as an integer) or
  9  * in percent. All elements with absolute height (i.e. in pixels) always will
 10  * have the given height. All "free" space (that is not filled with elements
 11  * with 'absolute' height) will be distributed among other elements in
 12  * proportion of their height percentage. Elements without 'height' in the
 13  * config will take equal portions of the "unallocated" height.</p>
 14  * <p>Supports panel collapsing, hiding, removal/addition. The adapter is provided
 15  * to use with Ext.SplitBar: <b>Ext.ux.layout.RowFitLayout.SplitAdapter</b>.</p>
 16  * <p>Example usage:</p>
 17  * <pre><code>
 18  var vp = new Ext.Viewport({
 19    layout: 'row-fit',
 20    items: [
 21      { xtype: 'panel', height: 100, title: 'Height in pixels', html: 'panel height = 100px' },
 22      { xtype: 'panel', height: "50%", title: '1/2', html: 'Will take half of remaining height' },
 23      { xtype: 'panel', title: 'No height 1', html: 'Panel without given height' },
 24      { xtype: 'panel', title: 'No height 2', html: 'Another panel' }
 25    ]
 26  });
 27  * </code></pre>
 28  * Usage of the split bar adapter:
 29  * <pre><code>
 30  var split = new Ext.SplitBar("elementToDrag", "elementToSize", Ext.SplitBar.VERTICAL, Ext.SplitBar.TOP);
 31  // note the Ext.SplitBar object is passed to the adapter constructor to set
 32  // correct minSize and maxSize:
 33  split.setAdapter(new Ext.ux.layout.RowFitLayout.SplitAdapter(split));
 34  * </code></pre>
 35  */
 36 
 37 Ext.ux.layout.RowFitLayout = Ext.extend(Ext.layout.ContainerLayout, {
 38   // private
 39   monitorResize: true,
 40 
 41   // private
 42   trackChildEvents: ['collapse', 'expand', 'hide', 'show'],
 43 
 44   // private
 45   renderAll: function(ct, target) {
 46     Ext.ux.layout.RowFitLayout.superclass.renderAll.apply(this, arguments);
 47     // add event listeners on addition/removal of children
 48     ct.on('add', this.containerListener);
 49     ct.on('remove', this.containerListener);
 50   },
 51 
 52   // private
 53   renderItem: function(c, position, target) {
 54     Ext.ux.layout.RowFitLayout.superclass.renderItem.apply(this, arguments);
 55 
 56     // add event listeners
 57     for (var i=0, n = this.trackChildEvents.length; i < n; i++) {
 58       c.on(this.trackChildEvents[i], this.itemListener);
 59     }
 60     c.animCollapse = false; // looks ugly together with row-fit layout
 61 
 62     // store some layout-specific calculations
 63     c.rowFit = {
 64       hasAbsHeight: false, // whether the component has absolute height (in pixels)
 65       relHeight: 0, // relative height, in pixels (if applicable)
 66       calcRelHeight: 0, // calculated relative height (used when element is resized)
 67       calcAbsHeight: 0 // calculated absolute height
 68     };
 69 
 70     // process height config option
 71     if (c.height) {
 72       // store relative (given in percent) height
 73       if (typeof c.height == "string" && c.height.indexOf("%")) {
 74         c.rowFit.relHeight = parseInt(c.height);
 75       }
 76       else { // set absolute height
 77         c.setHeight(c.height);
 78         c.rowFit.hasAbsHeight = true;
 79       }
 80     }
 81   },
 82 
 83   // private
 84   onLayout: function(ct, target) {
 85     Ext.ux.layout.RowFitLayout.superclass.onLayout.call(this, ct, target);
 86 
 87     if (this.container.collapsed || !ct.items || !ct.items.length) { return; }
 88 
 89     // first loop: determine how many elements with relative height are there,
 90     // sums of absolute and relative heights etc.
 91     var absHeightSum = 0, // sum of elements' absolute heights
 92         relHeightSum = 0, // sum of all percent heights given in children configs
 93         relHeightRatio = 1, // "scale" ratio used in case sum <> 100%
 94         relHeightElements = [], // array of elements with 'relative' height for the second loop
 95         noHeightCount = 0; // number of elements with no height given
 96 
 97     for (var i=0, n = ct.items.length; i < n; i++) {
 98       var c = ct.items.itemAt(i);
 99 
100       if (!c.isVisible()) { continue; }
101 
102       // collapsed panel is treated as an element with absolute height
103       if (c.collapsed) { absHeightSum += c.getFrameHeight(); }
104       // element that has an absolute height
105       else if (c.rowFit.hasAbsHeight) {
106         absHeightSum += c.height;
107       }
108       // 'relative-heighted'
109       else {
110         if (!c.rowFit.relHeight) { noHeightCount++; } // element with no height given
111         else { relHeightSum += c.rowFit.relHeight; }
112         relHeightElements.push(c);
113       }
114     }
115 
116     // if sum of relative heights <> 100% (e.g. error in config or consequence
117     // of collapsing/removing panels), scale 'em so it becomes 100%
118     if (noHeightCount == 0 && relHeightSum != 100) {
119       relHeightRatio = 100 / relHeightSum;
120     }
121 
122     var freeHeight = target.getStyleSize().height - absHeightSum, // "unallocated" height we have
123         absHeightLeft = freeHeight; // track how much free space we have
124 
125     while (relHeightElements.length) {
126       var c = relHeightElements.shift(), // element we're working with
127           relH = c.rowFit.relHeight * relHeightRatio, // height of this element in percent
128           absH = 0; // height in pixels
129 
130       // no height in config
131       if (!relH) {
132         relH = (100 - relHeightSum) / noHeightCount;
133       }
134 
135       // last element takes all remaining space
136       if (!relHeightElements.length) { absH = absHeightLeft; }
137       else { absH = Math.round(freeHeight * relH / 100); }
138 
139       // anyway, height can't be negative
140       if (absH < 0) { absH = 0; }
141 
142       c.rowFit.calcAbsHeight = absH;
143       c.rowFit.calcRelHeight = relH;
144 
145       c.setHeight(absH);
146       absHeightLeft -= absH;
147     }
148 
149   },
150 
151 
152   /**
153    * Event listener for container's children
154    * @private
155    */
156   itemListener: function(item) {
157     item.ownerCt.doLayout();
158   },
159 
160 
161   /**
162    * Event listener for the container (on add, remove)
163    * @private
164    */
165   containerListener: function(ct) {
166     ct.doLayout();
167   }

174   /**
175    * @param {Ext.SplitBar} splitbar to which adapter is applied.
176    *   If supplied, will set correct minSize and maxSize.
177    */
178   Ext.ux.layout.RowFitLayout.SplitAdapter = function(splitbar) {
179     if (splitbar && splitbar.el.dom.nextSibling) {
180       var next = Ext.getCmp( splitbar.el.dom.nextSibling.id ),
181           resized = Ext.getCmp(splitbar.resizingEl.id);
182 
183       if (next) {
184         splitbar.maxSize = (resized.height || resized.rowFit.calcAbsHeight) +
185                            next.getInnerHeight() - 1; // seems can't set height=0 in IE, "1" works fine
186       }
187       splitbar.minSize = resized.getFrameHeight() + 1;
188     }
189   }

191   Ext.extend(Ext.ux.layout.RowFitLayout.SplitAdapter, Ext.SplitBar.BasicLayoutAdapter, {
192 
193     setElementSize: function(splitbar, newSize, onComplete) {
194       var resized = Ext.getCmp(splitbar.resizingEl.id);
195 
196       // can't resize absent, collapsed or hidden panel
197       if (!resized || resized.collapsed || !resized.isVisible()) return;
198 
199       // resizingEl has absolute height: just change it
200       if (resized.rowFit.hasAbsHeight) {
201         resized.setHeight(newSize);
202       }
203       // resizingEl has relative height: affects next sibling
204       else {
205         if (splitbar.el.dom.nextSibling) {
206           var nextSibling = Ext.getCmp( splitbar.el.dom.nextSibling.id ),
207               deltaAbsHeight = newSize - resized.rowFit.calcAbsHeight, // pixels
208               nsRf = nextSibling.rowFit, // shortcut
209               rzRf = resized.rowFit,
210               // pixels in a percent
211               pctPxRatio = rzRf.calcRelHeight / rzRf.calcAbsHeight,
212               deltaRelHeight = pctPxRatio * deltaAbsHeight; // change in height in percent
213 
214           rzRf.relHeight = rzRf.calcRelHeight + deltaRelHeight;
215 
216           if (nsRf.hasAbsHeight) {
217             var newHeight = nextSibling.height - deltaAbsHeight;
218             nextSibling.height = newHeight;
219             nextSibling.setHeight(newHeight);
220           }
221           else {
222             nsRf.relHeight = nsRf.calcRelHeight - deltaRelHeight;
223           }
224         }
225       }
226       // recalculate heights
227       resized.ownerCt.doLayout();
228     } // of setElementSize
