  5 /** It instantiates a new DateOperator.
  6   * @class This is the general class for all DateOperators (with the exception of DatePeriodOperator)<BR>
  7   * <B>NB:</B> The value of a date operator (ie: value[0].value) will be always in the following format: Y-m-d H:i:s
  8   * @constructor
  9   * @extends Ext.ux.netbox.core.Operator
 10   * @param {String} id The id of the operator
 11   * @param {String} label The label of the operator
 12   * @param {String} format The format of the date. The date format should be divided from time format by a space.
 13   * If you provided only the date format, the time field will not appear.<br>
 14   * Supported formats:
 15   * <PRE>
 16   * Format  Description                                                               Example returned values
 17   * ------  -----------------------------------------------------------------------   -----------------------
 18   * d       Day of the month, 2 digits with leading zeros                             01 to 31
 19   * m       Numeric representation of a month, with leading zeros                     01 to 12
 20   * Y       A full numeric representation of a year, 4 digits                         Examples: 1999 or 2003
 21   * y       A two digit representation of a year                                      Examples: 99 or 03
 22   * H       24-hour format of an hour with leading zeros                              00 to 23
 23   * i       Minutes, with leading zeros                                               00 to 59
 24   * s       Seconds, with leading zeros                                               00 to 59
 25   *</PRE>
 26   */

 46   createEditor: function(operatorId){
 47     var editor;
 48     var splittedFormat=this.format.split(" ");

 50       var dateTimeField=new Ext.ux.form.DateTime({

 62     }else{
 63       editor=new Ext.ux.netbox.date.DateTextEditor(new Ext.form.DateField({
 64                 format: splittedFormat[0],
 65                 allowBlank: false
 66               }),
 67               {format: this.format}
 68             );
 69     }
 70     return editor;

 73   /** This function controls if the given value is an array with at least an element. If the given element is an object
 74     * with the format {label: , value: } and the value is a valid date in the format Y-m-d H:i:s, it returns an array containing the first element
 75     * otherwise it returns an empty array
 76     * @param {Object} value The value to convert
 77     * @return {Array} The converted object
 78     */
 79   convertValue: function(value){
 80     if(value !==null && value !== undefined && Ext.type(value)=="array"){
 81       if(value.length>0 && value[0].value!== undefined && value[0].label!== undefined){
 82         if(this.getField().checkDate(value[0].label) && this.getField().checkDate(value[0].value,'Y-m-d H:i:s')){
 83           if(value.length==1){
 84             return(value);
 85           } else {
 86             return([value[0]]);
 87           }
 88         }
 89       }
 90     }
 91     return([]);
 92   },

 95     * @return {String} The format of the dates
 96     */
 97   getFormat : function(){
 98     return this.format;
 99   }

112 Ext.extend(Ext.ux.netbox.date.DateRangeEditor,Ext.ux.netbox.FilterEditor,/** @scope Ext.ux.netbox.date.DateRangeEditor.prototype */{
113   /** This method parses the values returned by the RangeField, and casts the values to follow the Y-m-d H:i:s format
114     * @return {Array} Array of object {label: value: } where value is always in the format Y-m-d H:i:s or it is an empty string if the value entered by the user is not a valid date
115     */
116   getValue: function(){
117     var val=Ext.ux.netbox.date.DateRangeEditor.superclass.getValue.call(this);
118     var toReturn=[];
119     for(var i=0; val && i < val.length; i++){
120       var date=Date.parseDate(val[i].value,this.format);
121       if(!date){
122         toReturn.push({label:"",value:""});
123         continue;
124       }
125       val[i].value=date.format('Y-m-d H:i:s');
126       toReturn.push(val[i]);
127     }
128     return(toReturn);
129   }
130 });

132 /** It instantiates a new DateRangeOperator
133   * @class This is the class that implements the range operator between 2 dates. The id of the operator is DATE_RANGE
134   * @constructor
135   * @extends Ext.ux.netbox.date.DateOperator
136   * @param {String} format See format parameter of DateField
137   */
138 Ext.ux.netbox.date.DateRangeOperator = function(format) {
139   Ext.ux.netbox.date.DateRangeOperator.superclass.constructor.call(this,"DATE_RANGE",this.includeText,format);
140   this.mapping={
141     d: '99',
142     m: '99',
143     Y: '9999',
144     y: '99',
145     H: '99',
146     i: '99',
147     s: '99'
148   }
149   var validateFn=function(value){
150     var isOk=this.getField().emptyNotAllowedFn(value);
151     if(isOk!==true){
152       return(isOk);
153     }
154     if(value.length!=2){
155       return(this.bothFromAndToNotEmpty);
156     }
157     var fromADate=this.getField().checkDate(value[0].value,'Y-m-d H:i:s');
158     var toADate=this.getField().checkDate(value[1].value,'Y-m-d H:i:s');
159     if(!fromADate && !toADate){
160       return(this.toAndFromNotADate);
161     }
162 
163     if(!fromADate){
164       return(this.fromNotADate);
165     }
166 
167     if(!toADate){
168       return(this.toNotADate);
169     }
170 
171     if(Date.parseDate(value[0].value,'Y-m-d H:i:s')>Date.parseDate(value[1].value,'Y-m-d H:i:s')){
172       return(this.fromBiggerThanTo);
173     }
174     return(true);
175   }
176   this.setValidateFn(validateFn);
177 }

251   /** This method, given the format, returns a mask to use
252     * in the InputTextMask for the given format
253     * @return {String} The format to use
254     */
255   calculateMask: function(){
256 	  var maskTmp='';
257     for(var i=0; i<this.format.length;i++){
258       if(this.mapping[this.format.charAt(i)]){
259         maskTmp+=this.mapping[this.format.charAt(i)];
260       }else{
261         maskTmp+=this.format.charAt(i);
262       }
263     }
264     return(maskTmp);

268 /** It creates a new DatePeriodOperator
269   * @class This is the class that implements the period operator. The id of the operator is DATE_PERIOD
270   * The available periods are the following:
271   * <PRE>
272   *   LAST_YEAR: between now and now -1 year
273   *   LAST_MONTH: between now and now -1 month
274   *   LAST_WEEK: between now and now -1 week
275   *   LAST_DAY: between now and now -1 day
276   *   LAST_HOUR: between now and now -1 hour
277   *   LAST_QUARTER: between now and now -15 minutes
278   * </PRE>
279   * @constructor
280   * @extends Ext.ux.netbox.core.Operator
281   * @param {String} format See format parameter of DateField
282   */
283 Ext.ux.netbox.date.DatePeriodOperator = function() {
284   Ext.ux.netbox.date.DatePeriodOperator.superclass.constructor.call(this,"DATE_PERIOD",this.periodText);
285   /** Store used to show the available values
286     * @property {Ext.data.Store} periodStore
287     * @private
288     */
289   this.periodStore=new Ext.data.SimpleStore({fields: ['value', 'label'],
290       data: [
291         ["LAST_QUARTER",this.quarterText],
292         ["LAST_HOUR",this.hourText],
293         ["LAST_DAY",this.dayText],
294         ["LAST_WEEK",this.weekText],
295         ["LAST_MONTH",this.monthText],
296         ["LAST_YEAR",this.yearText]
297       ]});
298    var validateFn=function(value){
299      if(this.getField().emptyNotAllowedFn(value)!==true){
300        return(this.getField().emptyNotAllowedFn(value));
301      }
302      if(value[0].value!=="LAST_QUARTER" && value[0].value!=="LAST_HOUR" && value[0].value!=="LAST_DAY"
303        && value[0].value!=="LAST_WEEK" && value[0].value!=="LAST_MONTH" && value[0].value!=="LAST_YEAR"){
304        return(this.valueNotExpected);
305      }
306      return(true);
307    }
308    this.setValidateFn(validateFn);
309 }
