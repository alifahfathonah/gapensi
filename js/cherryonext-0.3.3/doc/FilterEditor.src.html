  3 Ext.namespace('Ext.ux.netbox');
  4 
  5 /** For a description of the parameters look at the documentation of Ext.Editor
  6   * @class This class extends Editor to allow an object as value (i.e. objects that String() translate into [object Object]. The 2 changes rows 
  7   * are documented using this comment: <pre>//<-- changed</pre>
  8   * @constructor
  9   */
 10 Ext.ux.netbox.FilterEditor = function(field,config){
 11   Ext.ux.netbox.FilterEditor.superclass.constructor.call(this,field,config);
 12 };
 13 
 14 Ext.extend(Ext.ux.netbox.FilterEditor,Ext.grid.GridEditor,/** @scope Ext.ux.netbox.FilterEditor.prototype */
 15 {
 16   /** This is absolutely equal to the method of GridEditor, but it calls this.setValue and not this.field.setValue
 17     */
 18   startEdit : function(el, value){
 19     if(this.editing){
 20         this.completeEdit();
 21     }
 22     this.boundEl = Ext.get(el);
 23     var v = value !== undefined ? value : this.boundEl.dom.innerHTML;
 24     if(!this.rendered){
 25         this.render(this.parentEl || document.body);
 26     }
 27     if(this.fireEvent("beforestartedit", this, this.boundEl, v) === false){
 28         return;
 29     }
 30     this.startValue = v;
 31     this.setValue(v);//<-- changed
 32     if(this.autoSize){
 33       var sz = this.boundEl.getSize();
 34       switch(this.autoSize){
 35         case "width":
 36         this.setSize(sz.width,  "");
 37         break;
 38         case "height":
 39         this.setSize("",  sz.height);
 40         break;
 41         default:
 42         this.setSize(sz.width,  sz.height);
 43       }
 44     }
 45     this.el.alignTo(this.boundEl, this.alignment);
 46     this.editing = true;
 47     this.show();
 48   },
 49 
 50   /** This is absolutely equal to the method of GridEditor, but it calls Ext.util.JSON.encode and not String to compare the values
 51     */
 52   completeEdit : function(remainVisible){
 53     if(!this.editing){
 54       return;
 55     }
 56     var v = this.getValue();
 57     if(this.revertInvalid !== false && !this.field.isValid()){
 58       v = this.startValue;
 59       this.cancelEdit(true);
 60     }
 61     if(Ext.util.JSON.encode(v) === Ext.util.JSON.encode(this.startValue) && this.ignoreNoChange){//<-- changed
 62       this.editing = false;
 63       this.hide();
 64       return;
 65     }
 66     if(this.fireEvent("beforecomplete", this, v, this.startValue) !== false){
 67       this.editing = false;
 68       if(this.updateEl && this.boundEl){
 69         this.boundEl.update(v);
 70       }
 71       if(remainVisible !== true){
 72         this.hide();
 73       }
 74       this.fireEvent("complete", this, v, this.startValue);
 75     }
 76   }
 77 });
