  5 /** It creates a new enumerator field
  6   * @param {String} id The id of the field
  7   * @param {String} label Optional. The label of the field. If not supplied the id is used.
  8   * @constructor
  9   * @extends Ext.ux.netbox.core.Field
 10   * @class This is the class that implements the field to use if the type is enumerator.
 11   * It contains as default the following operators:
 12   * <ul>
 13   *   <li> STRING_EQUAL </li>
 14   *   <li> STRING_DIFFERENT </li>
 15   * </ul>
 16   * The default operator is STRING_EQUAL.
 17   * The STRING_LIST and STRING_NOT_LIST operations should be manually added if needed.
 18   */
 19 Ext.ux.netbox.string.EnumField = function(id,label) {
 20   Ext.ux.netbox.string.EnumField.superclass.constructor.call(this,id,label);
 21   var equalOperator = new Ext.ux.netbox.core.Operator("STRING_EQUAL",this.stringEqualsLabel);
 22   this.addOperator(equalOperator);
 23   this.setDefaultOperator(equalOperator);
 24   this.addOperator(new Ext.ux.netbox.core.Operator("STRING_DIFFERENT",this.stringDifferentLabel));
 25 }
 26 
 27 Ext.extend(Ext.ux.netbox.string.EnumField,Ext.ux.netbox.core.Field,/** @scope Ext.ux.netbox.string.EnumField.prototype */{
 28 
 29   stringEqualsLabel: "=",
 30   stringDifferentLabel: "!=",
 31 
 32     /** Label of the STRING_LIST operation
 33     * @property
 34     */
 35   stringListText: "in",
 36   /** Label of the STRING_NOT_IN_LIST operation
 37     * @property
 38     */
 39   stringNotListText: "not in"
 40 
 41 });
