  3 Ext.namespace('Ext.ux.netbox.core');
  4 
  5 /** This constructor instantiates an editor with an Ext.ux.Andrie.Select to use as editor widget
  6   * @class This class is used to edit a value of a elementary filter when a set of available values must be shown.
  7   * The values MUST be in the format [{label:..., value:...}]
  8   * @constructor
  9   * @param {Ext.data.Store} store The store used to retrieve the data.
 10   * It must have 2 columns, label (which will be used as displayField) and value (which will be used as labelField).
 11   * @config {Object} config See the parameter with the same name of Ex.ux.FilterEditor.
 12   * Also were added in the config 4 boolean parameters: remote, forceReload,  multiSelect and caseSensitive.
 13   * config.remote: true if the store is remote, false if the data is local (ie, if store is a Ext.data.SimpleStore). Dafault is false.
 14   * config.forceReload: true if the store is to be reloaded everytime the combo expands. Dafault is false.
 15   * config.multiSelect: true if more than one selection is allowed, false for 1. Dafault is false.
 16   * config.caseSensitive: true if the value should be compared with the store's one with case sensitive. Default is true.
 17   * @extends Ext.ux.netbox.FilterEditor
 18   */
 19 
 20 Ext.ux.netbox.core.AvailableValuesEditor=function(store,config){
 21 
 22   if(config==undefined){
 23     config={};
 24   }
 25   var mode='local';
 26   if(config.remote==true)
 27     mode='remote';
 28   if(config.multiSelect==undefined){
 29     config.multiSelect=false;
 30   }
 31   this.fieldCombo=new Ext.ux.Andrie.Select({
 32     store         : store,
 33     displayField  : 'label',
 34     valueField    : 'value',
 35     selectOnFocus : true,
 36     mode          : mode,
 37     triggerAction : 'all',
 38     selectOnFocus : true,
 39     typeAhead     : true,
 40     multiSelect   : config.multiSelect,
 41     minChars      : 0
 42   });
 43   if(!config.multiSelect){
 44     this.fieldCombo.on('select',this.completeEditLater,this);
 45   }
 46   if(config.forceReload){
 47     this.fieldCombo.on("beforequery",function(qe){ qe.combo.lastQuery = null; });
 48   }
 49   if(config.caseSensitive)
 50     this.caseSensitive=true;
 51   else
 52     this.caseSensitive=false;
 53   Ext.ux.netbox.core.AvailableValuesEditor.superclass.constructor.call(this,this.fieldCombo,config);
 54   this.store=store;
 55 }
 56 
 57 Ext.extend(Ext.ux.netbox.core.AvailableValuesEditor,Ext.ux.netbox.FilterEditor,/** @scope Ext.ux.netbox.core.AvailableValuesEditor.prototype */{
 58 
 59   /** This method sets the value. This means that it sets the value of the inner combo.
 60     * If the value is an array with at least one element, the value to set is the array of the given elements.
 61     * Otherwise the value will be the empty string.
 62     * @param {Object} value The value to set.
 63     */
 64   setValue: function(value){
 65     var val=[];
 66     var rawVal=[];
 67     var label='';
 68     if(value!=undefined && value!=null && Ext.type(value)=="array"){
 69       if(value.length>0){
 70         for(var i=0; i< value.length && ((this.fieldCombo.multiSelect) || i<1); i++){
 71           val.push(value[i].value);
 72           if(value[i].label!=undefined){
 73             rawVal.push(value[i].label);
 74           } else {
 75             rawVal.push(value[i].value);
 76           }
 77         }
 78       }
 79     }
 80     this.originalValue=value;
 81 
 82     Ext.ux.netbox.core.AvailableValuesEditor.superclass.setValue.call(this,val);
 83     /* Hack to show the right label even if the store is not loaded.*/
 84     Ext.form.ComboBox.superclass.setValue.call(this.fieldCombo,rawVal.join(","));
 85     this.fieldCombo.value=val;
 86     this.fieldCombo.rawValueArray=rawVal;
 87   },
 88   
 89   /** This is a hack. If I stop editing on the select event, the gridpanel will scroll to the first row if there is a scrollbar.
 90     * The reason is that the ComboBox will request the focus after the event, even if it's not visible 
 91     * (the editor that contains the combo is already hidden)
 92     * I delay the complete of the editing at the end of the browser event queue (0 milliseconds of delay), to avoid the problem
 93     * @provate
 94     * @ignore
 95     */
 96   completeEditLater: function(){
 97     var task=new Ext.util.DelayedTask(this.completeEdit,this);
 98     task.delay(0);
 99   },
100   
101   /** This method gets the value. It searches the values in the store to have, with the values, the labels. If the store is not loaded,
102     * since the user didn't modify the value, it returns the original value.
103     * @return {Array} An array ob objects in the format {label:..., value:...}
104     */
105   getValue: function() {
106     var val=Ext.ux.netbox.core.AvailableValuesEditor.superclass.getValue.call(this);
107     if(Ext.type(val)=='string'){
108       val=val.split(',');
109     }
110     var toReturn=[];
111     for(var i=0; i<val.length;i++){
112       var j=this.store.find('value',val[i],0,false,this.caseSensitive);
113       if(j<0)
114         continue;
115       var record=this.store.getAt(j);
116       toReturn.push({label: record.get('label'), value: val[i]});
117     }
118     //if the user clicks on the field and then it presses Enter, the store is not loaded...
119     if((val.length>0 && val[0]!=="") && toReturn.length==0)
120       return(this.originalValue);
121     else
122       return toReturn;
123   }
124 });
